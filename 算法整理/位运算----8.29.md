+ 原地操作
+ 因为每个数可以待的位置是可以确定的，只不过不能破坏原来的数据
+ 因为数据范围小，可以直接利用高位对数据存储，到时候所有的数据直接右移十位，得到最终的目标数组。
+ **位运算一定注意加上括号**



```cpp
class Solution {
public:
    vector<int> shuffle(vector<int>& nums, int n) {
        int si = nums.size();
        for (int i = 0; i < nums.size(); ++i)
        {
            int j = i >= n ? (2 * (i - n) + 1) : 2 * i;
            nums[j] = nums[j] + ((nums[i] & 1023) << 10);
        }
        for (auto &c : nums)
        {
            c = c >> 10;
        }
        return nums;
    }
};
```
